// Função que retorna a aresta que pertence a árvore mínima e pertence as arestas de b e faz parte do caminho da árvore saindo de a
Busca(a, b)
for each w in Adj[a]
    if b == w and (PertenceT(a, b) or PertenceT(b, a))
         return (a, b)
    Busca(w, b)
